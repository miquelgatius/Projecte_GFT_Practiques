
/*Creació de la base de dades*/
CREATE DATABASE funator;


/*Seleccionar DB*/
use funator;

/*Creació de la taula d'usuaris*/
create table usuaris(
user_id varchar(20) primary key not null,
pass varchar(32) not null,
acc_name varchar(20) not null,
email varchar(35) not null UNIQUE check(email like  '%_@__%.__%'),
enabled boolean not null default true
) ENGINE=InnoDB;


/*Creació de la taula província*/
create table provincia(
prov_id int not null primary key AUTO_INCREMENT,
nom varchar(40) not null
) ENGINE=InnoDB;


/*Creació de la taula localitat*/
create table localitat(
local_id int not null primary key AUTO_INCREMENT,
prov_id int not null  references provincia(prov_id),
nom varchar(40) not null
) ENGINE=InnoDB;

/*Creació de la taula empresa*/
create table empresa(
CIF varchar(9) not null,
nom varchar(30) not null,
local int not null,
carrer varchar(40) not null,
telefon varchar(13),
password varchar(50),
primary key(CIF,carrer),
constraint foreign key(local) references localitat(local_id)
on update cascade
on delete restrict
) ENGINE=InnoDB;


/*Creació de la taula d'events*/
CREATE TABLE events(
event_id int not null primary key AUTO_INCREMENT,
nom_event varchar(80) not null,
preu decimal(10,2) not null,
data date not null,
descripcio varchar(1000),
imatge blob,
CIF varchar(9) not null,
constraint foreign key(CIF) references empresa(CIF)
on update cascade
on delete restrict
) ENGINE=InnoDB;

/*Creació de la taula ofertes*/
create table ofertes(
oferta_id int not null primary key AUTO_INCREMENT,
nom varchar(40),
descripcio varchar(200) not null,
preu decimal (10,2) not null,
CIF varchar(9) not null,
data_inici date not null,
data_final date not null,
constraint foreign key(CIF) references empresa(CIF)
on update cascade
on delete restrict
 )ENGINE=InnoDB;

/*Creació de la taula participants*/
create table participants(
event_id int not null references events(event_id),
user_id varchar(20) not null,
primary key(event_id,user_id),
constraint foreign key(event_id) references events(event_id)
on update cascade
on delete restrict,
constraint foreign key(user_id) references usuaris(user_id)
on update cascade
on delete restrict
)ENGINE=InnoDB;

/*Creació de la taula d'empresa-usuaris*/
create table empresausuari(
user_id varchar(20) not null,
CIF varchar(9) not null,
primary key (user_id,CIF),
constraint foreign key(CIF) references empresa(CIF)
on update cascade
on delete restrict,
constraint foreign key(user_id) references usuaris(user_id)
on update cascade
on delete restrict
)ENGINE=InnoDB;
